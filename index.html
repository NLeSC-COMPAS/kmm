KMM

KMM is a lightweight C++ middleware for accelerated computing, with native support for CUDA.

Highlights of KMM:

  • KMM manages the application memory

    • Allocations of host and device memory is automated

    • Data transfers to and from device are transparently performed when necessary

  • No need to rewrite your kernels

    • KMM can schedule and execute native C++ and CUDA functions

Basic Example

This example shows how to run a CUDA kernel implementing a vector add operation with KMM.

 1#include <deque>
 2#include <iostream>
 3
 4#include "spdlog/spdlog.h"
 5
 6#include "kmm/kmm.hpp"
 7
 8#define SIZE 65536000
 9
10__global__ void vector_add(const float* A, const float* B, float* C, int size) {
11    int item = (blockDim.x * blockIdx.x) + threadIdx.x;
12
13    if (item < size) {
14        C[item] = A[item] + B[item];
15    }
16}
17
18void initialize(float* A, float* B) {
19#pragma omp parallel for
20    for (unsigned int item = 0; item < SIZE; item++) {
21        A[item] = 1.0;
22        B[item] = 2.0;
23    }
24
25    std::cout << "initialize\n";
26}
27
28void verify(const float* C) {
29#pragma omp parallel for
30    for (unsigned int item = 0; item < SIZE; item++) {
31        if (fabsf(C[item] - 3.0f) > 1.0e-9) {
32            std::cout << "ERROR" << std::endl;
33            break;
34        }
35    }
36
37    std::cout << "SUCCESS" << std::endl;
38}
39
40int main(void) {
41    spdlog::set_level(spdlog::level::debug);
42
43    unsigned int threads_per_block = 256;
44    unsigned int n_blocks = ceil((1.0 * SIZE) / threads_per_block);
45    int n = SIZE;
46
47    // Create manager
48    auto manager = kmm::build_runtime();
49    std::deque<kmm::EventId> events;
50
51    for (size_t i = 0; i < 20; i++) {
52        // Request 3 memory areas of a certain size
53        auto A = kmm::Array<float>(n);
54        auto B = kmm::Array<float>(n);
55        auto C = kmm::Array<float>(n);
56
57        // Initialize array A and B on the host
58        manager.submit(kmm::Host(), initialize, write(A), write(B));
59
60        // Execute the function on the device.
61        manager.submit(kmm::GPUKernel(n_blocks, threads_per_block), vector_add, A, B, write(C), n);
62
63        // Verify the result on the host.
64        auto verify_event = manager.submit(kmm::Host(), verify, C);
65
66        events.push_back(verify_event);
67        if (events.size() >= 5) {
68            manager.wait(events[events.size() - 5]);
69        }
70    }
71
72    manager.synchronize();
73    std::cout << "done\n";
74
75    return 0;
76}
